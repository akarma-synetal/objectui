## Schema

```plaintext
interface SonnerSchema {
  action: 'sonner';
  message: string;                // Toast message
  description?: string;           // Additional description
  type?: 'default' | 'success' | 'error' | 'warning' | 'info';
  duration?: number;              // Auto-close duration (ms)
  
  // Action
  action?: {
    label: string;
    onClick: string | ActionConfig;
  };
}
```
