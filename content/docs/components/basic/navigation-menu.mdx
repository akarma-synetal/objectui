## Schema

```plaintext
interface NavigationMenuItem {
  label: string;
  href?: string;
  description?: string;
  icon?: string;
  items?: NavigationMenuItem[];   // Submenu items
}

interface NavigationMenuSchema {
  type: 'navigation-menu';
  items: NavigationMenuItem[];    // Menu items
  className?: string;
}
```
