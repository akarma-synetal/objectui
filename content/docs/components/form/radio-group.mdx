## Schema

```plaintext
interface RadioOption {
  value: string;
  label: string;
  disabled?: boolean;
}

interface RadioGroupSchema {
  type: 'radio-group';
  options: RadioOption[];         // Available options
  value?: string;                 // Selected value
  
  // Layout
  direction?: 'vertical' | 'horizontal';
  
  // Events
  onValueChange?: string | ActionConfig;
  
  // States
  disabled?: boolean;
  
  // Styling
  className?: string;
}
```
