## Schema

```plaintext
interface InputSchema {
  type: 'input';
  name?: string;
  label?: string;
  inputType?: 'text' | 'password' | 'email' | 'number' | 'tel' | 'url';
  placeholder?: string;
  defaultValue?: string | number;
  required?: boolean;
  disabled?: boolean;
  readonly?: boolean;
}
```
