## Schema

```plaintext
interface InputOTPSchema {
  type: 'input-otp';
  length: number;                 // Number of digits
  value?: string;                 // Current value
  separator?: boolean;            // Show visual separator
  
  // Events
  onChange?: string | ActionConfig;
  onComplete?: string | ActionConfig;
  
  // States
  disabled?: boolean;
  
  // Styling
  className?: string;
}
```
