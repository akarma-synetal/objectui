## Schema

```plaintext
interface BreadcrumbItem {
  label: string;
  href?: string;
  icon?: string;
}

interface BreadcrumbSchema {
  type: 'breadcrumb';
  items: BreadcrumbItem[];        // Breadcrumb items
  separator?: string;             // Custom separator
  className?: string;
}
```
