## Schema

```plaintext
interface ContextMenuItem {
  label?: string;
  value?: string;
  icon?: string;
  type?: 'separator';
  disabled?: boolean;
}

interface ContextMenuSchema {
  type: 'context-menu';
  trigger: ComponentSchema;       // Trigger element
  items: ContextMenuItem[];       // Menu items
  onSelect?: string | ActionConfig;
  className?: string;
}
```
