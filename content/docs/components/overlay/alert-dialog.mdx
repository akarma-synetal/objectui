## Schema

```plaintext
interface AlertDialogSchema {
  type: 'alert-dialog';
  title: string;                // Dialog title
  description: string;           // Dialog description
  
  // Trigger
  trigger: ComponentSchema;      // Component that triggers the dialog
  
  // Actions
  actions?: ComponentSchema[];   // Action buttons
  
  // Styling
  className?: string;
}
```
