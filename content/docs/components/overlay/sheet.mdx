## Schema

interface SheetSchema {
  type: 'sheet';

  title?: string;                 // Sheet title
  description?: string;           // Sheet description
  content: ComponentSchema;       // Sheet content
  side?: 'left' | 'right' | 'top' | 'bottom';
  className?: string;
}
