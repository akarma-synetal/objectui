## Schema

```plaintext
interface DropdownMenuItem {
  label?: string;
  value?: string;
  icon?: string;
  variant?: 'default' | 'destructive';
  type?: 'separator';
  disabled?: boolean;
}

interface DropdownMenuSchema {
  type: 'dropdown-menu';
  trigger: ComponentSchema;       // Trigger component
  items: DropdownMenuItem[];      // Menu items
  
  // Events
  onSelect?: string | ActionConfig;
  
  // Styling
  className?: string;
}
```
